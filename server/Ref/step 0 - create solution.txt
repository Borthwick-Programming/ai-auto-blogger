# üìÅ Create solution in working folder:
dotnet new sln -n WorkflowEngine

# üß± Create core backend projects
# The Web API template includes launchSettings.json automatically
dotnet new webapi -n WorkflowEngine.Api --no-https   # optional flag to skip HTTPS setup
dotnet new classlib -n WorkflowEngine.Domain
dotnet new classlib -n WorkflowEngine.Runtime
dotnet new classlib -n WorkflowEngine.Infrastructure

# ‚ûï Add projects to solution
dotnet sln WorkflowEngine.sln add WorkflowEngine.Api/WorkflowEngine.Api.csproj
dotnet sln WorkflowEngine.sln add WorkflowEngine.Domain/WorkflowEngine.Domain.csproj
dotnet sln WorkflowEngine.sln add WorkflowEngine.Runtime/WorkflowEngine.Runtime.csproj
dotnet sln WorkflowEngine.sln add WorkflowEngine.Infrastructure/WorkflowEngine.Infrastructure.csproj

# üîó Add project references
dotnet add WorkflowEngine.Api reference WorkflowEngine.Domain
dotnet add WorkflowEngine.Api reference WorkflowEngine.Runtime
dotnet add WorkflowEngine.Api reference WorkflowEngine.Infrastructure

dotnet add WorkflowEngine.Runtime reference WorkflowEngine.Domain
dotnet add WorkflowEngine.Infrastructure reference WorkflowEngine.Domain
